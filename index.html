11位带有*号的数字还原纯数字，个人编程水平较低，望理解

<!DOCTYPE html>
<html>
<head>
  <meta charset="UTF-8">
  <style>
    #result {
      margin-top: 10px;
      font-weight: bold;
    }

    #copyButton {
      margin-top: 10px;
    }
  </style>
</head>
<body>
  <h2>输入</h2>
  <label for="countInput">请输入数字的个数：</label>
  <input type="number" id="countInput">
  <br>
  <label for="numbersInput">请输入11位*加数字的组合（用逗号分隔）：</label>
  <input type="text" id="numbersInput">
  <br>
  <button onclick="generateNumbers()">生成</button>

  <h2>输出</h2>
  <div id="result"></div>
  <button id="copyButton" onclick="copyResult()">复制结果</button>

  <script>
    function generateNumbers() {
      var count = parseInt(document.getElementById('countInput').value);
      var numbersInput = document.getElementById('numbersInput').value;
      var numbers = numbersInput.split(',');

      var result = [];

      // 遍历每个11位数字组合
      for (var i = 0; i < numbers.length; i++) {
        var number = numbers[i];
        var asteriskIndex = number.indexOf('*');
        var newNumberPrefix = number.slice(0, asteriskIndex);
        var newNumberSuffix = number.slice(asteriskIndex + 1);

        // 将*的位置替换成0到9的数字
        for (var j = 0; j < 10; j++) {
          var newNumber = newNumberPrefix + j + newNumberSuffix;
          result.push(newNumber);
        }
      }

      // 显示结果
      var resultDiv = document.getElementById('result');
      resultDiv.innerHTML = result.join('<br>');

      // 显示复制按钮
      var copyButton = document.getElementById('copyButton');
      copyButton.style.display = 'block';
    }

    function copyResult() {
      var resultDiv = document.getElementById('result');
      var range = document.createRange();
      range.selectNodeContents(resultDiv);
      window.getSelection().removeAllRanges();
      window.getSelection().addRange(range);
      document.execCommand('copy');
      window.getSelection().removeAllRanges();

      alert('结果已复制到剪贴板！');
    }
  </script>
</body>
</html>
